<p style="color:#F00;"><strong>新约 - 罗马书(Romans) 第2章</strong></p>

	</tr>
	<tr>

	</tr>
</table>

	<tr>

	<tr>

	</tr>
</table>

	<tr>
		<td width="73" align="center">英文：</td>

</table>
